不好从跑最小费用最大流入手，那就从构造最小费用最大流入手

首先构造最大流为 inf⁡\operatorname {inf}inf
建 n+1n + 1n+1 个点 0∼n0\sim n0∼n，对于 0≤i<n0\le i < n0≤i<n，代表从 iii 号点流向 i+1i + 1i+1 号点
连边 (S⁡,0,inf⁡,0),(n,T⁡,inf⁡,0)(\operatorname {S}, 0, \operatorname {inf}, 0), (n, \operatorname {T}, \operatorname {inf}, 0)(S,0,inf,0),(n,T,inf,0)，保证最大流跑出来肯定是 inf⁡\operatorname {inf}inf
接下来处理对于 1≤i≤n1\le i\le n1≤i≤n 的 iii 号点至少被 aia_iai​ 个区间覆盖的问题，那考虑让 inf⁡−ai\operatorname {inf} - a_iinf−ai​ 的流量可以 000 花费流过去，剩下的则必须通过购买区间流过去，连边 (i−1,i,inf⁡−ai)(i - 1, i, \operatorname {inf} - a_i)(i−1,i,inf−ai​)
最后来考虑购买区间的问题，那相当于对于 1≤i≤m1\le i\le m1≤i≤m，花费 cic_ici​ 直接跳过 [si,ti][s_i, t_i][si​,ti​]，连边 (si−1,ti,inf⁡,ci)(s_i - 1, t_i, \operatorname {inf}, c_i)(si​−1,ti​,inf,ci​)，需要注意的是因为点的定义，sis_isi​ 会 −1- 1−1，而 tit_iti​ 不会

# include <bits/stdc++.h>
using namespace std;
const int inf = 1e9;
const int N = 1e3 + 10, M = 2e4 + 10;
int Ev [M * 2], Ext [M * 2], Ew [M * 2], Ec [M * 2], Eead [N];
int Etot = 1;
void EaddL (int u, int v, int w, int c) {
	++ Etot, Ev [Etot] = v, Ew [Etot] = w, Ec [Etot] = c, Ext [Etot] = Eead [u], Eead [u] = Etot;
	return ;
}
void Eadd (int u, int v, int w, int c) {
	EaddL (u, v, w, c), EaddL (v, u, 0, - c);
	return ;
}
int Erst [N], dis [N], vis [N];
int S, T, n;
int SPFA () {
	for (int i = 0; i <= n; ++ i) {
		dis [i] = inf, vis [i] = 0, Erst [i] = Eead [i];
	}
	dis [S] = 0, vis [S] = 1;
	queue <int> q; 
	q .push (S);
	while (! q .empty ()) {
		int u = q .front ();
		q .pop ();
		vis [u] = 0;
		for (int i = Erst [u]; i; i = Ext [i]) {
			int v = Ev [i], w = Ew [i], c = Ec [i];
			if (w && dis [u] + c < dis [v]) {
				dis [v] = dis [u] + c;
				if (! vis [v]) {
					vis [v] = 1;
					q .push (v);
				}
			}
		}
	}
	return dis [T] < inf;
}
int allcost;
int Dinic (int u, int flow) {
	if (u == T) {
		return flow;
	}
	int rest = flow;
	vis [u] = 1;
	for (int i = Erst [u]; i && rest; i = Ext [i]) {
		Erst [u] = i;
		int v = Ev [i],  w = Ew [i], c = Ec [i];
		if (! vis [v] && w && dis [u] + c == dis [v]) {
			int k = Dinic (v, min (w, rest));
			if (! k) {
				dis [v] = inf;
			}
			allcost += k * c, rest -= k, Ew [i] -= k, Ew [i ^ 1] += k;
		}
	}
	vis [u] = 0;
	return flow - rest;
}
const int F1 = 1e3 + 10, F2 = 1e4 + 10;
int a [F1];
int s [F2], t [F2], c [F2];
int main () {
	int an, am;
	scanf ("%d %d", & an, & am);
	S = an + 1, T = n = an + 2;
	for (int i = 1; i <= an; ++ i) {
		scanf ("%d", & a [i]);
	}
	for (int i = 1; i <= am; ++ i) {
		scanf ("%d %d %d", & s [i], & t [i], & c [i]);
	}
	Eadd (S, 0, inf, 0), Eadd (an, T, inf, 0);
	for (int i = 1; i <= an; ++ i) {
		Eadd (i - 1, i, inf - a [i], 0);
	}
	for (int i = 1; i <= am; ++ i) {
		Eadd (s [i] - 1, t [i], inf, c [i]);
	}
	while (SPFA ()) {
		int flow;
		while ((flow = Dinic (S, inf)));
	}
	printf ("%d", allcost);
	return 0;
}
