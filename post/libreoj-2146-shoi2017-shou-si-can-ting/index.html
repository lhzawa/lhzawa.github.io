# include <bits/stdc++.h>
using namespace std;
const int inf = 1e8;
const int N = 11000 + 100, M = 44000 + 100;
int Ev [M * 2], Ext [M * 2], Ew [M * 2];
int Eead [N];
int Etot = 1;
void EaddL (int u, int v, int w) {
    Ev [++ Etot] = v, Ew [Etot] = w, Ext [Etot] = Eead [u], Eead [u] = Etot;
    return ;
}
void Eadd (int u, int v, int w) {
    EaddL (u, v, w), EaddL (v, u, 0);
    return ;
}
int dep [N], Erst [N];
int S, T;
int n;
int BFS () {
    for (int i = 1; i <= n; ++ i) {
        dep [i] = 0;
        Erst [i] = Eead [i];
    }
    dep [S] = 1;
    queue <int> q; 
    q .push (S);
    while (! q .empty ()) {
        int u = q .front ();
        q .pop ();
        for (int i = Erst [u]; i; i = Ext [i]) {
            if (Ew [i] && ! dep [Ev [i]]) {
                dep [Ev [i]] = dep [u] + 1;
                q .push (Ev [i]);
                if (Ev [i] == T) {
                    return 1;
                }
            }
        }
    }
    return 0;
}
int dinic (int u, int flow) {
    if (u == T) {
        return flow;
    }
    int rest = flow;
    for (int i = Erst [u]; i && rest; i = Ext [i]) {
        Erst [u] = i;
        if (Ew [i] && dep [u] + 1 == dep [Ev [i]]) {
            int k = dinic (Ev [i], min (Ew [i], rest));
            if (! k) {
                dep [Ev [i]] = 0;
            }
            rest -= k, Ew [i] -= k, Ew [i ^ 1] += k;
        }
    }
    return flow - rest;
}
const int F = 1000 + 10;
int an, am;
int id [F] [F];
int a [F];
int w [F] [F];
int main () {
    scanf ("%d %d", & an, & am);
    n = 1000;
    for (int i = 1; i <= an; ++ i) {
    	for (int j = i; j <= an; ++ j) {
    		++ n, id [i] [j] = n;
    	}
    }
    S = n + 1, T = n + 2;
    n += 2;
    for (int i = 1; i <= an; ++ i) {
    	scanf ("%d", & a [i]);
    }
    for (int i = 1; i <= an; ++ i) {
    	for (int j = i; j <= an; ++ j) {
    		scanf ("%d", & w [i] [j]);
    	}
    }
    for (int i = 1; i <= 1000; ++ i) {
    	Eadd (S, i, 0), Eadd (i, T, am * i * i);
    }
    int ans = 0;
    for (int i = 1; i <= an; ++ i) {
    	for (int j = i; j <= an; ++ j) {
    		if (i == j) {
    			w [i] [j] -= a [i];
    		}
    		if (w [i] [j] >= 0) {
    			ans += w [i] [j];
    			Eadd (S, id [i] [j], w [i] [j]), Eadd (id [i] [j], T, 0);
    		}
    		else {
    			Eadd (S, id [i] [j], 0), Eadd (id [i] [j], T, - w [i] [j]);
    		}
    		if (i < j) {
    			Eadd (id [i] [j], id [i + 1] [j], inf), Eadd (id [i] [j], id [i] [j - 1], inf);
    		}
    		else {
    			Eadd (id [i] [j], a [i], inf);
    		}
    	}
    }
    while (BFS ()) {
    	int flow;
    	while ((flow = dinic (S, inf))) {
    		ans -= flow;
    	}
    }
    printf ("%d", ans);
    return 0;
