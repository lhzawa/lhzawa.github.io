GYM 101173B Bipartite Blanket

Hall 定理直接应用

Hall 定理：
二分图中在同部图的点集 SSS 有完美匹配当且仅当对于 SSS 的子集 TTT，与 TTT 里的点有直接连边的点的集合 N(T)N(T)N(T)，有 ∣T∣≤∣N(T)∣|T|\le |N(T)|∣T∣≤∣N(T)∣

这部分可以直接对于左右部图状压求解每个点集是否有完美匹配

同时还有个性质：
若左部图有点集 SSS 满足有完美匹配，右部图有点集 TTT 满足有完美匹配，则 S+TS + TS+T 也有完美匹配

所以记录下左右部图有完美匹配的点集用二分或双指针求解即可

# include <bits/stdc++.h>
using namespace std;
const int N = 20, M = 1 << 20;
int mL [N], mR [N], wL [N], wR [N];
int tot1 [M], NL [M], NR [M], valL [M], valR [M], fL [M], fR [M];
vector <int> L, R;
int main () {
	int n, m;
	scanf ("%d %d", & n, & m);
	for (int i = 0; i < n; ++ i) {
		for (int j = 0; j < m; ++ j) {
			int c;
			scanf ("%1d", & c);
			if (c) {
				mL [i] |= (1 << j), mR [j] |= (1 << i);
			}
		}
	}
	for (int i = 0; i < n; ++ i) {
		scanf ("%d", & wL [i]);
	}
	for (int i = 0; i < m; ++ i) {
		scanf ("%d", & wR [i]);
	}
	int Mx;
	scanf ("%d", & Mx);
	for (int i = 0; i < max (1 << n, 1 << m); ++ i) {
		tot1 [i] = tot1 [i >> 1] + (i & 1);
	}
	for (int i = 0; i < (1 << n); ++ i) {
		fL [i] = 1;
		for (int j = 0; j < n; ++ j) {
			if ((i >> j) & 1) {
				fL [i] &= fL [i ^ (1 << j)], valL [i] += wL [j], NL [i] |= mL [j];
			}
		}
		fL [i] &= (tot1 [i] <= tot1 [NL [i]]);
		if (fL [i]) {
			L .push_back (valL [i]);
		}
	}
	for (int i = 0; i < (1 << m); ++ i) {
		fR [i] = 1;
		for (int j = 0; j < n; ++ j) {
			if ((i >> j) & 1) {
				fR [i] &= fR [i ^ (1 << j)], valR [i] += wR [j], NR [i] |= mR [j]; 
			}
		}
		fR [i] &= (tot1 [i] <= tot1 [NR [i]]);
		if (fR [i]) {
			R .push_back (valR [i]);
		}
	}
	sort (R .begin (), R .end ());
	long long ans = 0;
	for (int l : L) {
		ans += R .end () - lower_bound (R .begin (), R .end (), Mx - l);
	}
	printf ("%lld", ans);
	return 0;
}
