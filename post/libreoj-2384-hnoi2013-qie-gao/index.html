考虑限制 1⁡\operatorname {1}1，即对于 1≤i≤P,1≤j≤Q1\le i\le P, 1\le j\le Q1≤i≤P,1≤j≤Q 的点 (i,j)(i,j)(i,j)，只有一个 fi,jf_{i, j}fi,j​ 的情况
对于 1≤i≤P,1≤j≤Q1\le i\le P, 1\le j\le Q1≤i≤P,1≤j≤Q 的点 (i,j)(i, j)(i,j) 建 R−1R - 1R−1 个点 si,j,1∼R−1s_{i, j, 1\sim R - 1}si,j,1∼R−1​，令 si,j,0=S⁡,si,j,R=T⁡s_{i, j, 0} = \operatorname {S}, s_{i, j, R} = \operatorname{T}si,j,0​=S,si,j,R​=T
对节点进行连边形成一条链，即对于 1≤k≤R1\le k\le R1≤k≤R，连边 (si,j,k−1,si,j,k,vi,j,k)(s_{i, j, k - 1}, s_{i, j, k}, v_{i, j, k})(si,j,k−1​,si,j,k​,vi,j,k​)，若这条边被割掉了则说明 fi,j=kf _ {i, j} = kfi,j​=k
进一步发现可能会有一条链割掉多条边的情况，对于这种情况，同样对于 1≤k≤R1\le k\le R1≤k≤R，连边 (si,j,k,si,j,k−1,inf⁡)(s_{i, j, k}, s_{i, j, k - 1}, \operatorname {inf})(si,j,k​,si,j,k−1​,inf)
证明考虑从最小割定义入手，对于正常的状态 SSTT⁡\operatorname {SSTT}SSTT，这条链上则肯定会有 STST⁡\operatorname {STST}STST 这样的状态，答案多一个 inf⁡\operatorname {inf}inf 一定不优

考虑限制 2⁡\operatorname {2}2，即对于 1≤i,u≤P,1≤j,v≤Q,∣i−u∣+∣j−v∣=11\le i, u\le P, 1\le j, v\le Q, |i - u| + |j - v| = 11≤i,u≤P,1≤j,v≤Q,∣i−u∣+∣j−v∣=1 的点 (i,j),(u,v)(i,j), (u, v)(i,j),(u,v)，满足 ∣fi,j−fu,v∣≤d|f_{i, j} - f_{u, v}| \le d∣fi,j​−fu,v​∣≤d

# include <bits/stdc++.h>
using namespace std;
const int inf = 1e8;
const int N = 64000 + 10, M = 384000 + 10;
int Ev [M * 2], Ext [M * 2], Ew [M * 2];
int Eead [N];
int Etot = 1;
void EaddL (int u, int v, int w) {
    Ev [++ Etot] = v, Ew [Etot] = w, Ext [Etot] = Eead [u], Eead [u] = Etot;
    return ;
}
void Eadd (int u, int v, int w) {
    EaddL (u, v, w), EaddL (v, u, 0);
    return ;
}
int dep [N], Erst [N];
int S, T;
int n;
int BFS () {
    for (int i = 1; i <= n; ++ i) {
        dep [i] = 0;
        Erst [i] = Eead [i];
    }
    dep [S] = 1;
    queue <int> q; 
    q .push (S);
    while (! q .empty ()) {
        int u = q .front ();
        q .pop ();
        for (int i = Erst [u]; i; i = Ext [i]) {
            if (Ew [i] && ! dep [Ev [i]]) {
                dep [Ev [i]] = dep [u] + 1;
                q .push (Ev [i]);
                if (Ev [i] == T) {
                    return 1;
                }
            }
        }
    }
    return 0;
}
int dinic (int u, int flow) {
    if (u == T) {
        return flow;
    }
    int rest = flow;
    for (int i = Erst [u]; i && rest; i = Ext [i]) {
        Erst [u] = i;
        if (Ew [i] && dep [u] + 1 == dep [Ev [i]]) {
            int k = dinic (Ev [i], min (Ew [i], rest));
            if (! k) {
                dep [Ev [i]] = 0;
            }
            rest -= k, Ew [i] -= k, Ew [i ^ 1] += k;
        }
    }
    return flow - rest;
}
