LibreOJ 6008 「网络流 24 题」餐巾计划

费用最小，联想到最小费用最大流

对于 1≤i≤n1\le i\le n1≤i≤n 的第 iii 天，建 222 个节点 xi,yix_i, y_ixi​,yi​，xix_ixi​ 代表新餐巾，yiy_iyi​ 代表旧餐巾

情况 1⁡\operatorname {1}1，每天用 rir_iri​ 块新餐巾
转化为每天减少 rir_iri​ 块新餐巾，增加 rir_iri​ 块旧餐巾
对于 1≤i≤n1\le i\le n1≤i≤n，连边 (xi,T⁡,ri,0),(S⁡,yi,ri,0)(x_i, \operatorname {T}, r_i, 0),(\operatorname {S}, y_i, r_i, 0)(xi​,T,ri​,0),(S,yi​,ri​,0)

情况 2⁡\operatorname {2}2，购买新餐巾
对于 1≤i≤n1\le i\le n1≤i≤n，连边 (S⁡,xi,inf⁡,P)(\operatorname {S}, x_i, \operatorname {inf}, P)(S,xi​,inf,P)

情况 3⁡\operatorname {3}3，使用快洗或慢洗
由当前该天的旧餐巾连向 MMM 或 NNN 天后的新餐巾
对于 1≤i≤n−M1\le i\le n - M1≤i≤n−M，连边 (yi,xi+M,inf⁡,F)(y_i, x_{i + M}, \operatorname {inf}, F)(yi​,xi+M​,inf,F)
对于 1≤j≤n−N1\le j\le n - N1≤j≤n−N，连边 (yi,xi+N,inf⁡,S)(y_i, x_{i + N}, \operatorname {inf}, S)(yi​,xi+N​,inf,S)

情况 4⁡\operatorname {4}4，新毛巾存至下一天
则下一天新毛巾加上该天的新毛巾，对于 1≤i<n1\le i < n1≤i<n，连边 (xi,xi+1,inf⁡,0)(x_i, x_{i + 1}, \operatorname {inf}, 0)(xi​,xi+1​,inf,0)

# include <bits/stdc++.h>
using namespace std;
const int inf = INT_MAX;
const int N = 2e3 + 10, M = 6e4 + 10;
int Ev [M * 2], Ext [M * 2], Ew [M * 2], Ec [M * 2], Eead [N];
int Etot = 1;
void EaddL (int u, int v, int w, int c) {
	++ Etot, Ev [Etot] = v, Ew [Etot] = w, Ec [Etot] = c, Ext [Etot] = Eead [u], Eead [u] = Etot;
	return ;
}
void Eadd (int u, int v, int w, int c) {
	EaddL (u, v, w, c), EaddL (v, u, 0, - c);
	return ;
}
int Erst [N], dis [N], vis [N];
int S, T, n;
int SPFA () {
	for (int i = 1; i <= n; ++ i) {
		dis [i] = inf, vis [i] = 0, Erst [i] = Eead [i];
	}
	dis [S] = 0, vis [S] = 1;
	queue <int> q; 
	q .push (S);
	while (! q .empty ()) {
		int u = q .front ();
		q .pop ();
		vis [u] = 0;
		for (int i = Erst [u]; i; i = Ext [i]) {
			int v = Ev [i], w = Ew [i], c = Ec [i];
			if (w && dis [u] + c < dis [v]) {
				dis [v] = dis [u] + c;
				if (! vis [v]) {
					vis [v] = 1;
					q .push (v);
				}
			}
		}
	}
	return dis [T] < inf;
}
int allcost;
int Dinic (int u, int flow) {
	if (u == T) {
		return flow;
	}
	int rest = flow;
	vis [u] = 1;
	for (int i = Erst [u]; i && rest; i = Ext [i]) {
		Erst [u] = i;
		int v = Ev [i],  w = Ew [i], c = Ec [i];
		if (! vis [v] && w && dis [u] + c == dis [v]) {
			int k = Dinic (v, min (w, rest));
			if (! k) {
				dis [v] = inf;
			}
			allcost += k * c, rest -= k, Ew [i] -= k, Ew [i ^ 1] += k;
		}
	}
	vis [u] = 0;
	return flow - rest;
}
const int AN = 1e3 + 10;
int r [AN];
int main () {
	int an, aP, aM, aF, aN, aS;
	scanf ("%d %d %d %d %d %d", & an, & aP, & aM, & aF, & aN, & aS);
	for (int i = 1; i <= an; ++ i) {
		scanf ("%d", & r [i]);
	}
	S = an * 2 + 1, T = n = an * 2 + 2;
	for (int i = 1; i <= an; ++ i) {
		if (i < an) {
			Eadd (i, i + 1, inf, 0);
		}
		Eadd (S, i, inf, aP);
		Eadd (i, T, r [i], 0);
		Eadd (S, i + an, r [i], 0);
		if (i + aM <= an) {
			Eadd (i + an, i + aM, inf, aF);
		}
		if (i + aN <= an) {
			Eadd (i + an, i + aN, inf, aS);
		}
	}
	while (SPFA ()) {
		int flow;
		while ((flow = Dinic (S, inf)));
	}
	printf ("%d", allcost);
	return 0;
}
